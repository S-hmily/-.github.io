def distEclud(x,y):
    return np.sqrt(np.sum((x-y)**2))  # è®¡ç®—æ¬§æ°è·ç¦»

def randCent(dataSet,k):
    m,n = dataSet.shape
    centroids = np.zeros((k,n))
    for i in range(k):
        index = int(np.random.uniform(0,m)) #
        centroids[i,:] = dataSet[index,:]
    return centroids
 
# kå‡å€¼èšç±»
def KMeans(dataSet,k):
 
    m = np.shape(dataSet)[0]  #è¡Œçš„æ•°ç›®
    # ç¬¬ä¸€åˆ—å­˜æ ·æœ¬å±äºå“ªä¸€ç°‡
    # ç¬¬äºŒåˆ—å­˜æ ·æœ¬çš„åˆ°ç°‡çš„ä¸­å¿ƒç‚¹çš„è¯¯å·®
    clusterAssment = np.mat(np.zeros((m,2)))
    clusterChange = True
 
    # ç¬¬1æ­¥ åˆå§‹åŒ–centroids
    centroids = randCent(dataSet,k)
    while clusterChange:
        clusterChange = False
 
        # éå†æ‰€æœ‰çš„æ ·æœ¬ï¼ˆè¡Œæ•°ï¼‰
        for i in range(m):
            minDist = 100000.0
            minIndex = -1
 
            # éå†æ‰€æœ‰çš„è´¨å¿ƒ
            #ç¬¬2æ­¥ æ‰¾å‡ºæœ€è¿‘çš„è´¨å¿ƒ
            for j in range(k):

                distance = distEclud(centroids[j,:],dataSet[i,:])
                if distance < minDist:
                    minDist = distance
                    minIndex = j
            # ç¬¬ 3 æ­¥ï¼šæ›´æ–°æ¯ä¸€è¡Œæ ·æœ¬æ‰€å±çš„ç°‡
            if clusterAssment[i,0] != minIndex:
                clusterChange = True
                clusterAssment[i,:] = minIndex,minDist**2
        #ç¬¬ 4 æ­¥ï¼šæ›´æ–°è´¨å¿ƒ
        for j in range(k):
            pointsInCluster = dataSet[np.nonzero(clusterAssment[:,0].A == j)[0]]  # è·å–ç°‡ç±»æ‰€æœ‰çš„ç‚¹
            centroids[j,:] = np.mean(pointsInCluster,axis=0)   # å¯¹çŸ©é˜µçš„è¡Œæ±‚å‡å€¼
 
    print("Congratulations,cluster complete!")
    return centroids,clusterAssment
